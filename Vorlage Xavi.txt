struCard* CreateCard(const char* pFilm, double IMDBrating, int BoxOffice) {
struCard* pTmp = (struCard*)malloc(sizeof(struCard));
strcpy_s(pTmp->Titel, pFilm);
pTmp->BoxOffice = BoxOffice;
pTmp->IMDBrating = IMDBrating;
pTmp->pNext = NULL;
return pTmp;
}


struCard* AddCard(struCard* pStart, struCard* pNew) {
if (pStart == NULL) {
pStart = pNew;
pNew->pNext = NULL;
}
else {
struCard* pLast = pStart;
while (pLast->pNext != NULL) pLast = pLast->pNext;
pLast->pNext = pNew;
}
return pStart;
}
void Karten_print(struCard* pStart) {
for (struCard* pOutCard = pStart; pOutCard != NULL; pOutCard = pOutCard->pNext)
printf("Film: %-30s, BoxOffice = %iM, IMDBrating = %g\n", pOutCard->Titel, pOutCard->BoxOffice, pOutCard->IMDBrating);
}
